module.exports = (sequelize, Sequelize) => {
    var Record = sequelize.define(
        'record',
        {
            id: {
                type: Sequelize.INTEGER(10),
                autoIncrement: true,
                primaryKey: true,
                allowNull: false
            },
            name: {
                type: Sequelize.STRING(255),
                allowNull: false
            },
            katalog_number: {
                type: Sequelize.STRING(255),
                allowNull: true
            },
            release_date: {
                type: Sequelize.DATEONLY,
                allowNull: true
            },
            country_id: {
                type: Sequelize.INTEGER(10),
                allowNull: false
            },
            cover: {
                type: Sequelize.STRING(1024),
                allowNull: true
            },
            rating: {
                type: Sequelize.FLOAT,
                allowNull: true
            }
        });

    Record.associate = (models) =>  {
        Record.hasMany(models.record_artist, {
            foreignKey: 'record_id',
            onDelete: 'CASCADE',
            onUpdate: 'CASCADE',
            sourceKey: 'id'
        });

        Record.hasMany(models.songs, {
            foreignKey: 'record_id',
            onDelete: 'CASCADE',
            onUpdate: 'CASCADE',
            sourceKey: 'id'
        });

        Record.hasMany(models.user_collection, {
            foreignKey: 'record_id',
            onDelete: 'CASCADE',
            onUpdate: 'CASCADE',
            sourceKey: 'id'
        });

        Record.belongsToMany(models.artists, {
            through: 'extraartists',
            foreignKey: 'record_id',
            otherKey: 'artist_id'
        });

        Record.belongsTo(models.countries, {
            foreignKey: 'country_id'
        });
    };
    return Record;
};
